#!/bin/bash

# Function to check if a number is prime
is_prime() {
    local num=$1
    if (( num <= 1 )); then
        return 1
    fi
    for (( i = 2; i * i <= num; i++ )); do
        if (( num % i == 0 )); then
            return 1
        fi
    done
    return 0
}

# Read a number from the user
echo "Enter a number:"
read num

# Check if the number is prime
if is_prime "$num"; then
    echo "$num is a prime number."
else
    echo "$num is not a prime number."
fi
